# create venv (first time only)
* `python3 -m venv venv`

# install git and configure
* `sudo apt update`
* `sudo apt upgrade`
* `sudo apt install git`
* `git config --global user.email "duane.northcutt@gmail.com"`
* `git config --global user.name "J. Duane Northcutt"`
* `git clone git@github.com:jduanen/linuxTools.git

# activate venv and install packages
* `source ./venv/bin/activate`
* `python3 -m pip install -r requirements.txt.pi0`

# install and configure chrony client
  * `sudo apt update`
  * `sudo apt install chrony`
  * `sudo systemctl stop systemd-timesyncd`
  * `sudo systemctl disable systemd-timesyncd`
  * `sudo systemctl enable --now chronyd`
  * `sudo cp /etc/chrony/chrony.conf /etc/chrony/chrony.conf.orig`
  * `sudo ex /etc/chrony/chrony.conf`
    - comment out default pools
    - use Ubuntu machine as chrony server
      * `server 192.168.166.13 iburst`
    - allow network corrections if needed
      * `allow 192.168.166.0/24`
  * restart chronyd
    - `sudo systemctl restart chronyd`
