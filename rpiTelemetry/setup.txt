# create venv (first time only)
* `python3 -m venv venv`

# install git and configure
* `sudo apt update`
* `sudo apt upgrade`
* `sudo apt install git`
* `git config --global user.email "duane.northcutt@gmail.com"`
* `git config --global user.name "J. Duane Northcutt"`
* `git clone git@github.com:jduanen/linuxTools.git

# activate venv and install packages
* `source ./venv/bin/activate`
* `python3 -m pip install -r requirements.txt`

# install and configure chrony client
* `sudo apt update`
* `sudo apt install chrony`
* `sudo systemctl stop systemd-timesyncd`
* `sudo systemctl disable systemd-timesyncd`
* `sudo systemctl enable --now chronyd`
* `sudo cp /etc/chrony/chrony.conf /etc/chrony/chrony.conf.orig`
* `sudo ex /etc/chrony/chrony.conf`
  - comment out default pools
  - use Ubuntu machine as chrony server
    * `server 192.168.166.13 iburst`
  - allow network corrections if needed
    * `allow 192.168.166.0/24`
* restart chronyd
  - `sudo systemctl restart chronyd`

# configure and setup the rpiTelemetry service
* `cd ${HOME}/Code/linuxTools/rpiTelemetry/`
* `sudo cp etc/systemd/system/rpiTelemetry.service /etc/systemd/system/`
* `sudo systemctl daemon-reload`
* `sudo systemctl enable rpiTelemetry`
* `sudo systemctl start rpiTelemetry`

# install and enable rpiTelemetry service
* `cd ${HOME}/Code/AircraftAudioId/audioCapture`
* `sudo mkdir -p /etc/rpiTelemetry`
* `sudo ex /etc/rpiTelemetry/mqtt.env`
```
MQTT_USER=<user>
MQTT_PASSWORD=<passwd>
```
* `sudo chmod 600 /etc/rpiTelemetry/mqtt.env`
* `sudo chown jdn:jdn /etc/rpiTelemetry/mqtt.env`
* `sudo cp etc/systemd/system/rpiTelemetry.service /etc/systemd/system/`
* `sudo systemctl daemon-reload`
* `sudo systemctl enable rpiTelemetry`
* `sudo systemctl start rpiTelemetry`
* `sudo journalctl -u audioCapMonitor`  # check that it's publishing
